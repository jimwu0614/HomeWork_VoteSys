UPDATE





UPDATE `students`

UPDATE `students` 
SET `uni_id` = '915084' ,
    `seat_num`='11042';            





//會讓符合條件的資料表的欄位都是相同的結果
UPDATE `students` 
SET `major` = '商業貿易科' 
WHERE `major`='國際貿易科';

DELETE

DELETE FROM `students` WHERE `id` = 10;



//會造成全部符合條件的資料都刪除
DELETE FROM `students` WHERE `major`='商業貿易科';



//會讓整個資料表的資料都被刪除
DELETE FROM `students`;

SELECT

SELECT * FROM `students` WHERE `id` &lt; 20 ;

SELECT * FROM `students` WHERE `id` > 100 AND `id` < 200;

SELECT * FROM `students` WHERE `id` < 100 OR `id` > 200;




SELECT `id` as '序號',`name` as '姓名' 
FROM `students` 
WHERE `id`<100 OR `id`>200;

SELECT * FROM `students` WHERE `birthday` >= '1984-01-01';

SELECT IN

SELECT * FROM `students` WHERE `id` IN(1,11,21,31,41,51);

SELECT * FROM `students` WHERE `seat_num` IN('10101','10203','10511');

SELECT BETWEEN

SELECT * FROM `students` WHERE `id` BETWEEN 110 AND 130;





SELECT * 
FROM `students` 
WHERE `birthday` 
BETWEEN '1983-04-01' AND '1983-04-30';

SELECT ORDER BY

SELECT * FROM `students` WHERE `id`<=49 ORDER BY `id` DESC;

SELECT * FROM `students` WHERE `id`<=49 ORDER BY `major`;

SELECT * FROM `students` WHERE `id`<=49 ORDER BY `major`,`id` DESC;

SELECT GROUP BY

SELECT `major` FROM `students` GROUP BY `major`;






SELECT `major`,
       SUBSTRING(`seat_num`,1,3) as '班級' 
FROM `students` 
GROUP BY `major`,
          SUBSTRING(`seat_num`,1,3)

SELECT * FROM `students` WHERE `id`<=49 ORDER BY `major`,`id` DESC;







SELECT `major`,
       SUBSTRING(`seat_num`,1,3) as '班級' 
FROM `students` 
GROUP BY `major`,
         SUBSTRING(`seat_num`,1,3) 
ORDER BY `id`




SELECT SUBSTRING(`address`,1,3) as "縣市" 
FROM `students` 
GROUP BY SUBSTRING(`address`,1,3)




SELECT SUBSTRING(`secondary`,6,4) as "國中" 
FROM `students` 
GROUP BY SUBSTRING(`secondary`,6,4)

SELECT LIMIT

SELECT * FROM `students` LIMIT 10;

SELECT * FROM `students` LIMIT 10,10;

SELECT * FROM `students` ORDER BY `major` LIMIT 10;

SELECT * FROM `students` ORDER BY `major`,`id` LIMIT 10,10;

聚合函式

SELECT max(`seat_num`) as 'MAX' FROM `students`

SELECT min(`seat_num`) as 'MIN' FROM `students`

SELECT `major`, max(`seat_num`) as 'MAX' FROM `students` GROUP BY `major`

SELECT `major`, count(`seat_num`) as '人數' FROM `students` GROUP BY `major`

字串函式

SELECT CONCAT(`seat_num`,`name`) as '座號姓名' FROM `students` LIMIT 10

SELECT * FROM `students` WHERE SUBSTRING(`seat_num`,1,3)='101'






SELECT `major`,
       GROUP_CONCAT(`name`) as '學生們' 
FROM `students` 
WHERE SUBSTRING(`seat_num`,1,3)='101' 
GROUP BY `major`






SELECT `major`,
       GROUP_CONCAT(`name`) as '學生們' 
FROM `students` 
WHERE LEFT(`seat_num`,3)='101' 
GROUP BY `major`

日期時間函式






SELECT Month(`birthday`) as '出生月份', 
       count(*) as '人數'
FROM `students` 
GROUP BY MONTH(`birthday`) 
ORDER BY MONTH(`birthday`)





SELECT `name` ,
       FLOOR(DATEDIFF(CURRENT_DATE(),`birthday`)/365) AS '年齡' 
FROM `students`
LIMIT 10;

IF()









SELECT `name` ,
       `birthday`,
       FLOOR(DATEDIFF(CURRENT_DATE(),`birthday`)/365) AS '年齡' ,
       IF(DATEDIFF(CURRENT_DATE(),REPLACE(`birthday`,LEFT(`birthday`,4),LEFT(CURRENT_DATE(),4))) > 0 ,
           365-DATEDIFF(CURRENT_DATE(),REPLACE(`birthday`,LEFT(`birthday`,4),LEFT(CURRENT_DATE(),4)))
         , 0-DATEDIFF(CURRENT_DATE(),REPLACE(`birthday`,LEFT(`birthday`,4),LEFT(CURRENT_DATE(),4)))) as '下次生日天數'
 FROM `students` 
 LIMIT 10;

聯表查詢-form where

SELECT * FROM `students`,`major` WHERE `major`.`id`=`students`.`major`





SELECT `students`.*,
       `major`.`name` as 'major' 
FROM `students`,`major` 
WHERE `major`.`id`=`students`.`major`

增加到第三張資料表但沒有加WHERE條件的結果







SELECT `students`.`id`,
       `students`.`name`,
       `graduate_school`.`name` as '畢業國中' ,
       `status`.`status` as '畢業狀態'
FROM `students`,`graduate_school`,`status`
WHERE `students`.`graduate_at`=`graduate_school`.`id`

加上WHERE條件後的結果








SELECT `students`.`id`,
       `students`.`name`,
       `graduate_school`.`name` as '畢業國中' ,
       `status`.`status` as '畢業狀態'
FROM `students`,`graduate_school`,`status`
WHERE `students`.`graduate_at`=`graduate_school`.`id` && 
	  `students`.`status_code`=`status`.`code`

加上第四張表後的查詢語法













SELECT `students`.`id`,
       `students`.`name`,
       `students`.`birthday`,
       `students`.`uni_id`,
       `dept`.`name` as '科系',
       `graduate_school`.`name` as '畢業國中' ,
       `status`.`status` as '畢業狀態'
FROM `students`,`graduate_school`,`status`,`dept`
WHERE `students`.`graduate_at`=`graduate_school`.`id` && 
      `students`.`status_code`=`status`.`code` &&
      `students`.`dept`=`dept`.`id`

多表查詢再合併其他SQL語法








SELECT `dept`.`name` as "科系",
        GROUP_CONCAT(`students`.`name`) as "學生"
FROM `students`,`dept`,`class_student`
WHERE `students`.`dept`=`dept`.`id` && 
      `students`.`school_num`=`class_student`.`school_num` &&
      `class_student`.`class_code`=101
GROUP BY `dept`.`name`

聯表查詢-Out Join

inner join的語法






SELECT `dept`.`name`,
        count(*) as '人數' 
FROM `dept`,`students` 
WHERE `students`.`dept`=`dept`.`id` 
GROUP By `students`.`dept`

LEFT JOIN的語法








SELECT `dept`.`name`,
        count(`students`.`id`) as '人數' 
FROM `dept`
LEFT JOIN `students` 
       ON `students`.`dept`=`dept`.`id`
GROUP By `dept`.`name`

多表LEFT JOIN的語法















SELECT `students`.`id`,
       `students`.`name`,
       `students`.`birthday`,
       `students`.`uni_id`,
       `dept`.`name` as '科系',
       `graduate_school`.`name` as '畢業國中' ,
       `status`.`status` as '畢業狀態'
FROM `students`
LEFT JOIN `graduate_school` 
       ON `students`.`graduate_at`=`graduate_school`.`id`
LEFT JOIN `status` 
       ON `students`.`status_code`=`status`.`code`
LEFT JOIN `dept` 
       ON `students`.`dept`=`dept`.`id`

子查詢

子查詢在SELECT區段的用法







SELECT `class_code` as '班級',
        count("*") as "人數" ,
        (SELECT count(*) FROM `class_student`) as '總人數',
        count(*) / (SELECT count(*) FROM `class_student`) as '比例'
FROM   `class_student` 
GROUP BY `class_code`

使用原始資料表的在FROM區段的查詢 






SELECT `class_student`.`school_num` as '學號',
       `students`.`name`,
       `students`.`addr`
FROM `class_student`,`students` 
WHERE `class_student`.`school_num`=`students`.`school_num`

利用子查詢在FROM區段簡化來源資料表的用法










SELECT `class_student`.`school_num` as '學號',
       `A`.`name`,
       `A`.`addr`
FROM `class_student`,
      (SELECT `students`.`school_num`,
              `students`.`name`,
              `students`.`addr`  
       FROM `students`)A
WHERE `class_student`.`school_num`=`A`.`school_num`

子查詢在WHERE區段的用法












SELECT `A`.`班級`,`A`.`人數`
FROM (SELECT `class_code` as '班級',
             count("*") as "人數" 
      FROM   `class_student` 
      GROUP BY `class_code`)A
WHERE `A`.`人數` >= (SELECT AVG(`B`.`人數`) 
                    FROM (SELECT `class_code` as '班級',
                                 count("*") as "人數" 
                          FROM `class_student` 
                          GROUP BY `class_code`)B
                   )
